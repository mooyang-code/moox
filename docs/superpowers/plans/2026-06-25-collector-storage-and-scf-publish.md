# Collector Storage And SCF Publish Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Migrate the data collector from the removed `DataService.WriteRows` API to the current storage read/write APIs, add a high-level data-subject registration API, and repair the data-collector cloud-function package/publish flows for both the frontend console and `moox-cli`.

**Architecture:** Storage keeps `SubjectSymbol` as an internal metadata model, but public callers register collected objects through `RegisterDataSubject`. The collector becomes config-driven for `space_id`, `dataset_id`, subject bindings, and symbol record version, while cloud-function publishing reuses Control's existing upload-to-COS and SCF creation path.

**Tech Stack:** Go, tRPC-Go, protobuf generated by `trpc-open`, Cobra CLI, Vue/TypeScript frontend, Tencent Cloud SCF/COS SDK.

---

## Pre-Implementation Review

No blocker remains, but the implementation must handle these issues explicitly:

- Current collector code calls `/trpc.storage.data.DataService/WriteRows`; the current storage public write APIs are `/trpc.storage.access.AccessService/WriteTimeSeriesRows` and `/trpc.storage.access.AccessService/WriteRecordRows`.
- `SubjectSymbol` is already represented in storage and has store-level support. Its standalone public mutation RPC must no longer be the collector-facing API.
- `RegisterDataSubject` should maintain `Subject`, internal `SubjectSymbol`, and `DatasetSubject` bindings idempotently. `WriteRecordRows` remains separate and writes Binance symbol details with `version=latest`.
- Collector must read `space_id`, kline dataset IDs, symbol record dataset IDs, and bind dataset IDs from YAML, not hard-code them.
- Kline rows must use `key.data_time` only for candle time and must format it as RFC3339/RFC3339Nano.
- Existing collector symbol normalization uses internal symbols like `BTC-USDT`; Binance external symbols are `BTCUSDT`. The plan keeps `subject_id` and `record_id` as the internal symbol and writes `external_symbol` separately.
- `binance_swap_symbols` metadata is missing and must be seeded.
- `modules/collector/Makefile build-scf` currently mutates source config and zips only `main *.yaml`; the new package command must write generated YAML into the zip and include nested `sources/`.
- SCF runtime is already mostly user-controlled, but `CreateNode` still defaults runtime to `Go1` when missing. Keep this as fallback only, not as a lower-layer hard-code.
- Tencent provider types include `Handler`, but `CreateFunction` does not currently pass it to the SCF SDK request; this should be fixed so users can set SCF handler when needed.
- The active OpenSpec change `redesign-quant-data-protocols` points toward future `Instrument/InstrumentAlias` terminology. This plan fixes the current `Subject` model while keeping collector config and client code isolated enough for later renaming.

```go
func (s *Service) RegisterDataSubject(ctx context.Context, req *pb.RegisterDataSubjectReq) (*pb.RegisterDataSubjectRsp, error) {
	if req == nil || req.GetSpaceId() == "" || req.GetDataSourceId() == "" || req.GetExternalSymbol() == "" || req.GetSubject() == nil || req.GetSubject().GetSubjectId() == "" {
		return &pb.RegisterDataSubjectRsp{RetInfo: response.Error(pb.ErrorCode_INVALID_PARAM, errors.New("space_id, data_source_id, external_symbol and subject.subject_id are required"))}, nil
	}

	subject := proto.Clone(req.GetSubject()).(*pb.Subject)
	subject.SpaceId = req.GetSpaceId()
	if subject.Status == "" {
		subject.Status = "active"
	}
	createdSubject, err := s.metadata.UpsertSubject(ctx, subject)
	if err != nil {
		return &pb.RegisterDataSubjectRsp{RetInfo: response.Error(pb.ErrorCode_INVALID_PARAM, err)}, nil
	}

	_, err = s.metadata.UpsertSubjectSymbol(ctx, &pb.SubjectSymbol{
		SpaceId:        req.GetSpaceId(),
		SubjectId:      createdSubject.GetSubjectId(),
		DataSourceId:   req.GetDataSourceId(),
		ExternalSymbol: req.GetExternalSymbol(),
		Status:         "active",
	})
	if err != nil {
		return &pb.RegisterDataSubjectRsp{RetInfo: response.Error(pb.ErrorCode_INVALID_PARAM, err)}, nil
	}

	bindings := make([]*pb.DatasetSubject, 0, len(req.GetDatasetBindings()))
	for _, binding := range req.GetDatasetBindings() {
		if binding == nil || binding.GetDatasetId() == "" {
			return &pb.RegisterDataSubjectRsp{RetInfo: response.Error(pb.ErrorCode_INVALID_PARAM, errors.New("dataset_bindings.dataset_id is required"))}, nil
		}
		item := proto.Clone(binding).(*pb.DatasetSubject)
		item.SpaceId = req.GetSpaceId()
		item.SubjectId = createdSubject.GetSubjectId()
		if item.SubjectRole == "" {
			item.SubjectRole = "normal"
		}
		if item.Status == "" {
			item.Status = "active"
		}
		if _, err := s.metadata.GetDataset(ctx, item.GetSpaceId(), item.GetDatasetId()); err != nil {
			return &pb.RegisterDataSubjectRsp{RetInfo: response.Error(pb.ErrorCode_DATASET_NOT_FOUND, err)}, nil
		}
		created, err := s.metadata.BindDatasetSubject(ctx, item)
		if err != nil {
			return &pb.RegisterDataSubjectRsp{RetInfo: response.Error(pb.ErrorCode_INVALID_PARAM, err)}, nil
		}
		bindings = append(bindings, created)
	}

	return &pb.RegisterDataSubjectRsp{
		RetInfo:         response.Success("success"),
		Subject:         createdSubject,
		DatasetBindings: bindings,
	}, nil
}
```

```go
type Client struct {
	endpoint string
	auth     AuthInfo
	http     *http.Client
}

func NewClient(endpoint string, auth AuthInfo) *Client {
	return &Client{
		endpoint: strings.TrimRight(endpoint, "/"),
		auth:     auth,
		http:     &http.Client{Timeout: 30 * time.Second},
	}
}

func (c *Client) RegisterDataSubject(ctx context.Context, req RegisterDataSubjectRequest) error {
	req.AuthInfo = c.auth
	var rsp RegisterDataSubjectResponse
	return c.post(ctx, registerDataSubjectPath, req, &rsp)
}

func (c *Client) WriteTimeSeriesRows(ctx context.Context, rows []TimeSeriesRow) error {
	var rsp WriteTimeSeriesRowsResponse
	return c.post(ctx, writeTimeSeriesRowsPath, WriteTimeSeriesRowsRequest{AuthInfo: c.auth, Rows: rows}, &rsp)
}

func (c *Client) WriteRecordRows(ctx context.Context, rows []RecordRow) error {
	var rsp WriteRecordRowsResponse
	return c.post(ctx, writeRecordRowsPath, WriteRecordRowsRequest{AuthInfo: c.auth, Rows: rows}, &rsp)
}

func (c *Client) post(ctx context.Context, path string, reqBody any, rspBody interface{ GetRetInfo() RetInfo }) error {
	data, err := json.Marshal(reqBody)
	if err != nil {
		return fmt.Errorf("marshal storage request: %w", err)
	}
	req, err := http.NewRequestWithContext(ctx, http.MethodPost, c.endpoint+path, bytes.NewReader(data))
	if err != nil {
		return fmt.Errorf("create storage request: %w", err)
	}
	req.Header.Set("Content-Type", "application/json")
	resp, err := c.http.Do(req)
	if err != nil {
		return fmt.Errorf("send storage request: %w", err)
	}
	defer resp.Body.Close()
	var raw bytes.Buffer
	_, _ = raw.ReadFrom(resp.Body)
	if resp.StatusCode != http.StatusOK {
		return fmt.Errorf("storage http %d: %s", resp.StatusCode, raw.String())
	}
	if err := json.Unmarshal(raw.Bytes(), rspBody); err != nil {
		return fmt.Errorf("decode storage response: %w", err)
	}
	ret := rspBody.GetRetInfo()
	if ret.Code != 0 {
		return fmt.Errorf("storage error %d: %s", ret.Code, ret.Msg)
	}
	return nil
}
```

```go
	}
	defer out.Close()
	zw := zip.NewWriter(out)
	defer zw.Close()

	if err := addFile(zw, opts.MainPath, "main"); err != nil {
		return err
	}
	return filepath.WalkDir(configDir, func(path string, d os.DirEntry, err error) error {
		if err != nil || d.IsDir() {
			return err
		}
		rel, err := filepath.Rel(configDir, path)
		if err != nil {
			return err
		}
		return addFile(zw, path, filepath.ToSlash(rel))
	})
}

func addFile(zw *zip.Writer, src string, name string) error {
	in, err := os.Open(src)
	if err != nil {
		return err
	}
	defer in.Close()
	info, err := in.Stat()
	if err != nil {
		return err
	}
	header, err := zip.FileInfoHeader(info)
	if err != nil {
		return err
	}
	header.Name = name
	header.Method = zip.Deflate
	writer, err := zw.CreateHeader(header)
	if err != nil {
		return err
	}
	_, err = io.Copy(writer, in)
	return err
}

func prepareConfigDir(opts BuildSCFPackageOptions) (string, func(), error) {
	tmp, err := os.MkdirTemp("", "moox-collector-scf-*")
	if err != nil {
		return "", func() {}, err
	}
	cleanup := func() { _ = os.RemoveAll(tmp) }
	if err := copyDir(opts.ConfigDir, tmp); err != nil {
		cleanup()
		return "", func() {}, err
	}
	configPath := filepath.Join(tmp, "config.yaml")
	if err := patchYAMLFile(configPath, opts.Version, opts.Overrides); err != nil {
		cleanup()
		return "", func() {}, err
	}
	return tmp, cleanup, nil
}

func copyDir(src, dst string) error {
	return filepath.WalkDir(src, func(path string, d os.DirEntry, err error) error {
		if err != nil {
			return err
		}
		rel, err := filepath.Rel(src, path)
		if err != nil {
			return err
		}
		target := filepath.Join(dst, rel)
		if d.IsDir() {
			return os.MkdirAll(target, 0755)
		}
		return copyFile(path, target)
	})
}

func copyFile(src, dst string) error {
	if err := os.MkdirAll(filepath.Dir(dst), 0755); err != nil {
		return err
	}
	in, err := os.Open(src)
	if err != nil {
		return err
	}
	defer in.Close()
	out, err := os.Create(dst)
	if err != nil {
		return err
	}
	defer out.Close()
	_, err = io.Copy(out, in)
	return err
}

func patchYAMLFile(path string, version string, overrides map[string]string) error {
	data, err := os.ReadFile(path)
	if err != nil {
		return err
	}
	var root map[string]any
	if err := yaml.Unmarshal(data, &root); err != nil {
		return err
	}
	if version != "" {
		setYAMLPath(root, "system.version", version)
	}
	for key, value := range overrides {
		setYAMLPath(root, key, value)
	}
	next, err := yaml.Marshal(root)
	if err != nil {
		return err
	}
	return os.WriteFile(path, next, 0644)
}

func setYAMLPath(root map[string]any, dotted string, value string) {
	parts := strings.Split(dotted, ".")
	cur := root
	for _, part := range parts[:len(parts)-1] {
		next, ok := cur[part].(map[string]any)
		if !ok {
			next = map[string]any{}
			cur[part] = next
		}
		cur = next
	}
	cur[parts[len(parts)-1]] = value
}
```
